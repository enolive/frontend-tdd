# Example

```tsx
import { render, screen } from '@testing-library/react'
import App from './App'

describe('Todo App', () => {
 it('shows header', () => {
    render(<App />)

    const header = screen.getByRole('heading', { level: 1 })
    expect(header).toHaveTextContent('Todo App')
  })
})
```
